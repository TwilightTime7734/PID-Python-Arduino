  397:     def record_auto_session_sample(sample) -> None:
  398:         nonlocal auto_last_sample_s, auto_event_peak_delta, auto_event_response_delay_s
  399:         auto_last_sample_s = time.monotonic()
  400:         command = auto_active_command
  401:         if command is None:
  402:             return
  403:         axis_value = pulse_axis_value(sample, command.axis)
  404:         directed_delta = (axis_value - auto_event_baseline) * float(command.direction)
  405:         if directed_delta > auto_event_peak_delta:
  406:             auto_event_peak_delta = float(directed_delta)
  407:         if auto_event_response_delay_s is None:
  408:             threshold_deg = max(2.0, (command.force_us / 15.0) * 0.35)
  409:             if directed_delta >= threshold_deg:
  410:                 auto_event_response_delay_s = max(0.0, time.monotonic() - auto_event_start_s)

  412:     def draw_channel_output(index: int, value: int) -> None:
  413:         clamped = max(1000, min(2000, value))
  414:         canvas = channel_output_canvases[index]
  415:         fill_id = channel_output_fill_ids[index]
  416: 
  417:         left = 2.0
  418:         right = 94.0
  419:         center = (left + right) / 2.0
  420:         y1 = 3.0
  421:         y2 = 13.0
  422: 
  423:         if clamped < 1500:
  424:             ratio = (1500 - clamped) / 500.0
  425:             x = center - (center - left) * ratio
  426:             canvas.coords(fill_id, x, y1, center, y2)
  427:             canvas.itemconfig(fill_id, fill="#E38C8C")
  428:         elif clamped > 1500:
  429:             ratio = (clamped - 1500) / 500.0
  430:             x = center + (right - center) * ratio
  431:             canvas.coords(fill_id, center, y1, x, y2)
  432:             canvas.itemconfig(fill_id, fill="#94D98F")
  433:         else:
  434:             canvas.coords(fill_id, center, y1, center, y2)

  479:     def get_pulse_action(event: tk.Event) -> str:

  488:     def cancel_adjust_repeat() -> None:
  489:         nonlocal adjust_repeat_after_id, adjust_repeat_handler, adjust_repeat_index, adjust_repeat_delta
  490:         if adjust_repeat_after_id is not None:
  491:             try:
  492:                 root.after_cancel(adjust_repeat_after_id)
  493:             except Exception:
  494:                 pass
  495:             finally:
  496:                 adjust_repeat_after_id = None
  497:         adjust_repeat_handler = None
  498:         adjust_repeat_index = None
  499:         adjust_repeat_delta = 0

  501:     def schedule_adjust_repeat() -> None:
  502:         nonlocal adjust_repeat_after_id
  503:         if adjust_repeat_handler is None or adjust_repeat_index is None or adjust_repeat_delta == 0:
  504:             adjust_repeat_after_id = None
  505:             return
  506:         adjust_repeat_handler(adjust_repeat_index, adjust_repeat_delta)
  507:         adjust_repeat_after_id = root.after(ADJUST_REPEAT_INTERVAL_MS, schedule_adjust_repeat)
